Installation

dotnet add package SirvImage
using Sirv;

Quick Start

var url = new SirvImage("https://demo.sirv.com/image.jpg")
    .Resize(width: 300, height: 200)
    .Quality(80)
    .Format("webp")
    .ToUrl();

Constructor

ConstructorDescription
new SirvImage(string url)Create from full URL
new SirvImage(string baseUrl, string path)Create from base URL and path
 
var img = new SirvImage("https://demo.sirv.com/image.jpg");
var img = new SirvImage("https://demo.sirv.com", "/image.jpg");

Option Classes

TextOptions

Properties: Size, FontSize, FontFamily, FontStyle, FontWeight, Color, Opacity, OutlineWidth, OutlineColor, OutlineOpacity, OutlineBlur, BackgroundColor, BackgroundOpacity, Align, Position, PositionX, PositionY, PositionGravity.

new TextOptions { Color = "white", FontSize = 24, Position = "center" }

WatermarkOptions

Properties: Position, PositionX, PositionY, PositionGravity, ScaleWidth, ScaleHeight, ScaleOption, Rotate, Opacity, Layer, CanvasColor, CanvasOpacity.

new WatermarkOptions { Position = "center", Opacity = 50 }

CanvasOptions

Properties: Width, Height, Color, Position, Opacity, AspectRatio, BorderWidth, BorderHeight, BorderColor, BorderOpacity.

FrameOptions

Properties: Style, Color, Width, RimColor, RimWidth.

Text Overlay

Text(string content, TextOptions? options)

Add text overlay. Call multiple times for multiple layers.

img.Text("Hello", new TextOptions { Color = "white", FontSize = 24, Position = "center" })
   .Text("World", new TextOptions { Position = "south" })

Full Examples

var url = new SirvImage("https://demo.sirv.com", "/product.jpg")
    .Resize(width: 800, option: "fill")
    .Crop(type: "face")
    .Format("webp")
    .Quality(85)
    .Sharpen(20)
    .Watermark("/logo.png", new WatermarkOptions { Position = "southeast", Opacity = 40 })
    .Text("Sale!", new TextOptions { Color = "red", FontSize = 32, Position = "north" })
    .ToUrl();
